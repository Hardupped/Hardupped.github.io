为了帮你更直观地理解Java调用GPT-5-Codex有多简单，这里提供一个完整的使用OkHttp的示例，一个经典的HTTP请求：

java
import okhttp3.*;
import com.google.gson.*;

public class GPT5CodexDemo {
    // [[千聚ai官网](https://www.qianjuai.com/)](https://www.qianjuai.com/)的API地址
    private static final String BASE_URL = "https://www.qianjuai.com/v1/chat/completions";
    // 你的千聚ai API Key
    private static final String API_KEY = "your-qianjuai-api-key-here";
    
    public static void main(String[] args) throws Exception {
        OkHttpClient client = new OkHttpClient();
        
        String json = "{\n" +
                      "  \"model\": \"gpt-5-codex\",\n" +
                      "  \"messages\": [{\n" +
                      "    \"role\": \"user\",\n" +
                      "    \"content\": \"在Java中编写一个高效的单例模式实现\"\n" +
                      "  }],\n" +
                      "  \"max_tokens\": 1000\n" +
                      "}";
        
        RequestBody body = RequestBody.create(json, MediaType.get("application/json; charset=utf-8"));
        
        Request request = new Request.Builder()
            .url(BASE_URL)
            .addHeader("Authorization", "Bearer " + API_KEY)
            .post(body)
            .build();
        
        try (Response response = client.newCall(request).execute()) {
            System.out.println("GPT-5 Codex返回的代码：");
            System.out.println(response.body().string());
        }
    }
}
