- 检测图片中的特定元素（如logo、肤色、物体位置）

这里提供一个万能模板，你可以直接复制使用：

javascript
const OpenAI = require('openai');

const client = new OpenAI({
  apiKey: process.env.QIANJU_API_KEY || '你的Key',
  baseURL: 'https://www.qianjuai.com/v1'
});

async function visionAnalysis(imageUrl, prompt = '请描述这张图片') {
  const model = process.env.VISION_MODEL || 'Qwen-VL-Plus'; // 可动态切换
  const response = await client.chat.completions.create({
    model: model,
    messages: [
      {
        role: 'user',
        content: [
          { type: 'text', text: prompt },
          { type: 'image_url', image_url: { url: imageUrl } }
        ]
      }
    ],
    max_tokens: 500
  });
  return response.choices[0].message.content;
}
