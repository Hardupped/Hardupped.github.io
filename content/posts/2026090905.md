import com.theokanning.openai.completion.chat.*;

import java.time.Duration;
import java.util.ArrayList;
import java.util.List;

public class Claude接入示例 {

    public static void main(String[] args) {
        
        // 1. 配置API Key和Base URL
        String apiKey = "sk-你的千聚API密钥"; 
        String baseUrl = "https://www.qianjuai.com/v1"; // 核心：一键切换

        // 2. 构建客户端
        OpenAiService service = new OpenAiService(
                baseUrl,
                apiKey,
                Duration.ofSeconds(30)
        );

        // 3. 构造消息列表
        List<ChatMessage> messages = new ArrayList<>();
        messages.add(new ChatMessage(
                ChatMessageRole.SYSTEM.value(),
                "你是一名资深Java开发者，能用简洁的语言解释复杂的技术概念。"
        ));
        messages.add(new ChatMessage(
                ChatMessageRole.USER.value(),
                "请用Java代码演示如何使用Stream API处理集合。"
        ));

        // 4. 构造请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("claude-3-5-sonnet-20241022") // 模型名称
                .messages(messages)
                .maxTokens(2048)
                .temperature(0.7)
                .build();

        // 5. 发起非流式请求（最稳定）
        try {
            ChatCompletionResult result = service.createChatCompletion(request);
            String response = result.getChoices().get(0).getMessage().getContent();
            System.out.println("[千聚ai大模型聚合站](https://www.qianjuai.com/) 【Claude】 回复：\n" + response);
        } catch (Exception e) {
            System.err.println("请求失败，请检查API Key或网络环境: " + e.getMessage());
        }
    }
}
