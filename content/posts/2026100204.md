### 场景一：流式输出（Stream），聊天体验更丝滑

如果你的项目是做聊天机器人或者实时对话，流式输出几乎是必选项。几行代码就能启用：

javascript
const axios = require('axios');

async function streamChat() {
  const response = await axios.post(
    'https://www.qianjuai.com/v1/chat/completions',
    {
      model: 'glm-4.5',
      messages: [
        { role: 'user', content: '请用一句话解释量子纠缠，要求小学生能听懂' }
      ],
      stream: true           // 打开流式开关
    },
    {
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Type': 'application/json'
      },
      responseType: 'stream' // 关键：设为 stream 类型
    }
  );

  response.data.on('data', chunk => {
    const lines = chunk.toString().split('\n').filter(line => line.trim() !== '');
    for (const line of lines) {
      if (line.includes('[DONE]')) break;
      if (!line.startsWith('data: ')) continue;
      try {
        const jsonData = JSON.parse(line.replace('data: ', ''));
        const content = jsonData.choices[0]?.delta?.content || '';
        process.stdout.write(content); // 逐步打印内容
      } catch (e) {
        // 跳过解析失败的片段
      }
    }
  });
}

streamChat();
