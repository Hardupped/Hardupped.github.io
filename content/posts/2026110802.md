const OpenAI = require('openai');

const client = new OpenAI({
  apiKey: 'sk-你的千聚APIKey',
  baseURL: 'https://www.qianjuai.com/v1'
});

async function codeReview(codeSnippet) {
  const response = await client.chat.completions.create({
    model: 'o3',  // 使用 o3 满血版
    messages: [
      {
        role: 'user',
        content: `请审查以下 JavaScript 代码，找出所有潜在 bug、性能问题和安全漏洞，并给出修复建议：\n\n${codeSnippet}`
      }
    ]
  });
  return response.choices[0].message.content;
}

const myCode = `
function fetchData(id) {
  return fetch('/api/data/' + id).then(res => res.json());
}
`;

codeReview(myCode).then(review => {
  console.log('代码审查结果：');
  console.log(review);
}).catch(err => {
  console.error('审查失败:', err);
});
