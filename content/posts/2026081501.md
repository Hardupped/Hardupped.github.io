// index.js
import OpenAI from 'openai';
import dotenv from 'dotenv';
dotenv.config();

// 初始化客户端
const client = new OpenAI({
  apiKey: process.env.QIANJU_API_KEY, // 你的千聚API Key
  baseURL: 'https://www.qianjuai.com/v1' // 关键：切换base_url
});

// 测试函数 —— 使用o3-mini模型（兼顾性能与成本）
async function testAI() {
  try {
    const completion = await client.chat.completions.create({
      model: 'o3-mini', // 或 gpt-4o、claude-3.5-sonnet 等
      messages: [
        { role: 'system', content: '你是一个专业的AI助手。' },
        { role: 'user', content: '用中文写一段50字的文章开场白，关于AI技术如何改变开发者工作方式。' }
      ],
      temperature: 0.7,
      max_tokens: 200
    });

    console.log('[千聚api聚合平台](https://www.qianjuai.com/)响应结果：');
    console.log(completion.choices[0].message.content);
    console.log(`\nToken使用量: ${completion.usage?.total_tokens || 'N/A'}`);
    
  } catch (error) {
    console.error('调用失败：', error.message);
  }
}

testAI();
