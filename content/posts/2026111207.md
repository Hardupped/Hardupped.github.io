import io.github.openai.OpenAiClient;
import io.github.openai.OpenAiClientV1;
import io.github.openai.completion.ChatCompletionRequest;
import io.github.openai.completion.ChatCompletionResult;

public class Grok3MiniExample {

    public static void main(String[] args) {
        // 2. 关键一步：改 base_url 和 API Key
        // 原来：https://api.openai.com/v1
        // 现在：https://www.qianjuai.com/v1
        String baseUrl = "https://www.qianjuai.com/v1";
        String apiKey = "sk-sua-YourApiKey";  // 换成在千聚获取的 Key

        // 3. 创建客户端
        OpenAiClient client = OpenAiClientV1.create()
                .baseUrl(baseUrl)
                .apiKey(apiKey)
                .build();

        // 4. 创建并发送请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("grok-3-mini")  // 直接用模型 ID
                .addMessage("user", "请用 Java 写一段快速排序。")
                .build();

        ChatCompletionResult response = client.createChatCompletion(request);

        System.out.println(response.getChoices().get(0).getMessage().getContent());
    }
}
