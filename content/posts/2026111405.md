from chromadb.utils import embedding_functions

    api_base="https://www.qianjuai.com/v1"

# 3. 创建本地向量数据库

# 4. 将知识库文档添加到向量数据库
knowledge_dir = "./my_knowledge"
for i, filename in enumerate(os.listdir(knowledge_dir)):
    filepath = os.path.join(knowledge_dir, filename)
    if os.path.isfile(filepath):
        with open(filepath, 'r', encoding='utf-8') as f:
            content = f.read()
        # 简单分块（实际可以更精细）
        chunks = content.split('\n\n') # 按双换行分割

            collection.add(
                ids=[f"{filename}_{j}"],
                documents=[chunk],
                metadatas=[{"source": filename}]
            )

print("知识库构建完成！")

## 第三步：搭建OpenAI兼容的问答服务接口

知识库建好了，接下来你需要一个接口来回答用户的问题。这个接口将完全遵循OpenAI的 `/v1/chat/completions` 格式，这样所有兼容OpenAI的工具都能直接使用。

创建一个 `my_rag_api.py` 文件：

python
from flask import Flask, request, jsonify
from openai import OpenAI
import chromadb
from chromadb.utils import embedding_functions
import uuid

app = Flask(__name__)

# 初始化千聚客户端和向量数据库（与上一步相同）
client = OpenAI(api_key="YOUR_QIANJUAI_API_KEY", base_url="https://www.qianjuai.com/v1")
sentence_transformer_ef = embedding_functions.OpenAIEmbeddingFunction(api_key="YOUR_QIANJUAI_API_KEY", model_name="text-embedding-ada-002", api_base="https://www.qianjuai.com/v1")
chroma_client = chromadb.PersistentClient(path="./knowledge_base")
collection = chroma_client.get_collection(name="my_docs", embedding_function=sentence_transformer_ef)

@app.route('/v1/chat/completions', methods=['POST'])
def chat_completion():
    data = request.get_json()
    user_message = data['messages'][-1]['content'] # 获取用户最后一条消息

    # 1. 在知识库里检索最相关的文档
    results = collection.query(query_texts=[user_message], n_results=3)
    context = "\n".join(results['documents'][0])

    # 2. 构建完整的对话消息（包含上下文和用户问题）
    messages = [
        {"role": "system", "content": f"你是一个知识库问答助手，请基于以下上下文回答问题：\n{context}"},
        {"role": "user", "content": user_message}
    ]

    # 3. 调用千聚平台的大模型生成答案
    response = client.chat.completions.create(
        model="gpt-4o", # 或其他你需要的模型
        messages=messages,
        stream=False
    )

    # 4. 封装成OpenAI兼容的响应格式
    return jsonify({
        "id": f"chatcmpl-{uuid.uuid4().hex}",
        "object": "chat.completion",
        "created": int(uuid.uuid1().time),
        "model": "gpt-4o",
        "choices": [
            {
                "index": 0,
                "message": {
                    "role": "assistant",
                    "content": response.choices[0].message.content
                },
                "finish_reason": "stop"
            }
        ],
        "usage": {
            "prompt_tokens": response.usage.prompt_tokens,
            "completion_tokens": response.usage.completion_tokens,
            "total_tokens": response.usage.total_tokens
        }
    })

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8000)


运行这个脚本（`python my_rag_api.py`），你的OpenAI兼容的知识库问答服务就在 `http://localhost:8000` 上运行了。
