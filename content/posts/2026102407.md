## 代码实战：从 0 到 1 搭建 AI 中台

下面这个 Python 脚本就是一个“企业级 AI 中台接口”，它只管理一个密钥，但能按业务需求路由到不同模型。

python
import openai

# [千聚ai聚合站](https://www.qianjuai.com/) 提供的唯一 API Key
company_api_key = "your-qianju-api-key"
# 千聚中台的统一 base_url

openai.api_key = company_api_key
openai.base_url = base_url

def smart_model_router(business_type: str, user_input: str) -> str:
    """
    根据业务类型自动选择模型并返回回复。
    整个企业只管理一个密钥，多个模型统一调用。
    """
    routes = {
        "reasoning": {  # 复杂推理 -> GPT-5 Reasoning
            "model": "gpt-5-reasoning",
            "temperature": 0.3,
        },
        "creative": {   # 创意生成 -> GPT-5 Turbo
            "model": "gpt-5-turbo",
            "temperature": 0.8,
        },
        "vision": {     # 图片识别 -> GPT-5 Vision
            "model": "gpt-5-vision",
            "temperature": 0.1,
        },
        "cost_saving": { # 低成本任务 -> DeepSeek-R1
            "model": "deepseek-r1",
            "temperature": 0.7,
        },
    }

    config = routes.get(business_type, routes["cost_saving"])

    messages = [{
        "role": "user",
        "content": user_input
    }]

    response = openai.ChatCompletion.create(
        model=config["model"],
        messages=messages,
        temperature=config["temperature"]
    )
    return response.choices[0].message.content

# 企业内部调用示例
