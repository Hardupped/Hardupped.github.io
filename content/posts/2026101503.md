以下是一个完整的、可直接运行的Java类，它通过**[千聚ai中转站](https://www.qianjuai.com/)**向ERNIE-Bot发起一次聊天请求。

java
package com.example.api;

import java.io.IOException;
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;
import com.google.gson.Gson;
import com.google.gson.JsonObject;
import com.google.gson.JsonArray;

public class UnifiedAIClient {

    public static void main(String[] args) {
        // 配置
        String apiKey = "sk-你的千聚Key"; // 替换为你的API Key
        String modelName = "ERNIE-Bot";   // 替换为你想要的模型
        String userMessage = "用一段话介绍苏州";

        // 构建请求体
        Gson gson = new Gson();
        JsonObject messageObj = new JsonObject();
        messageObj.addProperty("role", "user");
        messageObj.addProperty("content", userMessage);
        
        JsonArray messagesArray = new JsonArray();
        messagesArray.add(messageObj);
        
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", modelName);
        requestBody.add("messages", messagesArray);
        
        // 发送请求
        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://www.qianjuai.com/v1/chat/completions"))
                .header("Authorization", "Bearer " + apiKey)
                .header("Content-Type", "application/json")
                .POST(BodyPublishers.ofString(gson.toJson(requestBody)))
                .build();
        
        try {
            HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
            // 解析并打印
            JsonObject responseJson = gson.fromJson(response.body(), JsonObject.class);
            String content = responseJson.getAsJsonArray("choices")
                              .get(0).getAsJsonObject()
                              .getAsJsonObject("message")
                              .get("content").getAsString();
            System.out.println("ERNIE-Bot回答: " + content);
        } catch (IOException | InterruptedException e) {
            e.printStackTrace();
        }
    }
}

你需要引入`com.google.code.gson`这个依赖。运行后，如果Key正确，网络畅通，你将在控制台看到文心一言给出的回答。
