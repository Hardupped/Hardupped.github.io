javascript
// 1. 引入 OpenAI 库

// 2. 初始化 OpenAI 客户端
const client = new OpenAI({
    // 【核心改动】只需要把 baseURL 换成[千聚api聚合站](https://www.qianjuai.com/)的统一地址！
    baseURL: 'https://www.qianjuai.com/v1',

    apiKey: 'sk-你的千聚API Key，请替换成实际密钥',
});

// 3. 定义一个异步函数来发送请求
async function callTongyi() {
    try {
        console.log('正在调用通义千问...');

        // 4. 创建聊天完成请求
        const response = await client.chat.completions.create({
            // 【核心改动】模型名称这里，填上通义千问的模型ID
            model: 'qwen-turbo',  // 可换成 qwen-plus, qwen-max 等
            messages: [
                { role: 'user', content: '你好，请用一句话介绍你自己。' }
            ],
            temperature: 0.7,
            max_tokens: 100,
        });

        // 5. 打印输出结果
        console.log('通义千问的回答是：');
        console.log(response.choices[0].message.content);

    } catch (error) {
        console.error('调用失败了，错误信息如下：');
        console.error(error);
    }
}

// 6. 执行函数

### 常见问题排查：真遇到报错也别慌

如果代码跑起来报错，99%是因为下面三个地方没写对：

1.  **API Key无效或过期**：检查`apiKey`字段是否填对了，Key前面有没有不小心多了个空格。可以回千聚后台重新生成一个试试。
2.  **模型名写错**：检查`model`字段。通义千问的模型ID中间是**短横线**（`-`），不是下划线（`_`），比如就是 `qwen-turbo`，不是 `qwen_turbo`。
3.  **网络不通**：检查你的`baseURL`是不是 `https://www.qianjuai.com/v1`，注意`https://`不能少，最后的`/v1`也不能少。

一般来说，解决掉上面3个问题，都能顺利跑通。
