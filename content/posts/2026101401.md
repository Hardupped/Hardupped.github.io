**2. 核心调用逻辑**

创建一个配置类 `Grok4Config.java`，这步最关键，它初始化了 HTTP 客户端。

java
import com.theokanning.openai.service.OpenAiService;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import java.time.Duration;

@Configuration
public class Grok4Config {

    @Value("${grok4.api-key}")
    private String apiKey;

    @Value("${grok4.base-url}")
    private String baseUrl;

    @Bean
    public OpenAiService openAiService() {
        // 关键：必须指定自定义的BaseUrl，指向千聚的网关
        return new OpenAiService(baseUrl, apiKey, Duration.ofSeconds(60));
    }
}


**3. 调用Grok-4进行推理**

现在直接注入我们的 `OpenAiService`，像调用GPT-4一样调用Grok-4。

java
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RestController;
import java.util.List;

@RestController
public class ChatController {

    @Autowired
    private OpenAiService openAiService;

    @PostMapping("/chat")
    public String chat(@RequestBody String userMessage) {
        // 组装消息
        List<ChatMessage> messages = List.of(
                new ChatMessage("system", "你是一个优秀的Java开发助手，精通Grok-4模型调用。"),
                new ChatMessage("user", userMessage)
        );

        // 创建请求，指定模型
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("grok-4") // 模型名直接写官方的，千聚自动识别！
                .messages(messages)
                .maxTokens(4096)
                .temperature(0.7)
                .build();

        // 发请求，拿结果
        return openAiService.createChatCompletion(request)
                .getChoices()
                .get(0)
                .getMessage()
                .getContent();
    }
}
