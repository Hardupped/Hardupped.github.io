def call_yiyan(model_name="ernie-4.0-8k", prompt="你好，请介绍一下你自己"):
    try:
        # 注意：千聚ai对文心一言的调用兼容Chat Completion接口
        response = client.chat.completions.create(

        )
        
        print("文心一言的回答：")
        for chunk in response:
            if chunk.choices[0].delta.content is not None:

        print("\n")
        
    except Exception as e:
        print(f"调用失败：{e}")

# 执行调用
if __name__ == "__main__":
    # 你可以在这里随意修改prompt
