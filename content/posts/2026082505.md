import com.openai.client.OpenAIClient;
import com.openai.client.okhttp.OpenAIOkHttpClientAsync;
import com.openai.models.chat.completions.ChatCompletion;
import com.openai.models.chat.completions.ChatCompletionCreateParams;
import com.openai.models.chat.completions.ChatCompletionMessageParam;
import com.openai.models.chat.completions.ChatCompletionUserMessageParam;

public class Grok4ThroughQianjuai {

    public static void main(String[] args) {
        // 1. 初始化客户端：只改base_url和api_key
        OpenAIClient client = OpenAIOkHttpClientAsync.builder()
            .baseUrl("https://www.qianjuai.com/v1")   // 关键！换成千聚的地址
            .apiKey("你的千聚API_KEY")                   // 千聚后台申请的key
            .build();

        // 2. 构造请求参数，model指定为grok-4
        ChatCompletionCreateParams params = ChatCompletionCreateParams.builder()
            .model("grok-4")
            .addUserMessage("请用Java写一个高效的斐波那契数列生成器，要求时间复杂度O(logn)")
            .maxTokens(2048)
            .temperature(0.7)
            .build();

        // 3. 发起调用，同步等待结果
        ChatCompletion completion = client.chat().completions().create(params);

        // 4. 输出结果
        String result = completion.choices().get(0).message().content();
        System.out.println("Grok4回复：");
        System.out.println(result);
    }
}
