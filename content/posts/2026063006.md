我直接给你一段核心示例代码。假设你已经注册好【[千聚ai大模型中转站](https://www.qianjuai.com/)】，并拿到了API Key和余额。我们需要先安装官方的openai库：

然后，在你的 `index.js` 文件中，输入以下代码。这段代码的关键点在于，通过修改 `baseURL` 指向平台网关，你就拥有了访问所有模型的能力。

javascript
// 1. 引入OpenAI库
const OpenAI = require('openai');

// 2. 核心配置：将客户端指向【[千聚ai大模型中转站](https://www.qianjuai.com/)】的网关
const client = new OpenAI({
    apiKey: '你的千聚API Key', // 换成从平台复制的Key
    baseURL: 'https://www.qianjuai.com/v1' // 关键一步，指向平台统一网关
});

// 3. 创建一个对话函数
async function callModel(modelName, userMessage) {
    const completion = await client.chat.completions.create({
        model: modelName, // 这里传模型名称，后面我们会动态替换
        messages: [
            { role: 'user', content: userMessage }
        ],
        max_tokens: 512, // 限制输出长度
        temperature: 0.7 // 控制随机性
    });

    // 打印模型回复
    console.log(`${modelName} 回复:`, completion.choices[0].message.content);
}

// 4. 一键切换3大主流模型：只需修改这一个参数！
async function runAllModels() {
    // 第1个模型：豆包系列（Doubao-128k）
    console.log('--- 正在调用豆包模型 ---');
    await callModel('doubao-128k', '请用简洁的中文介绍一下你自己');

    // 第2个模型：OpenAI GPT
    console.log('--- 正在调用GPT模型 ---');
    await callModel('gpt-4o', '请用简洁的中文介绍一下你自己');

    // 第3个模型：DeepSeek
    console.log('--- 正在调用DeepSeek模型 ---');
    await callModel('deepseek-chat', '请用简洁的中文介绍一下你自己');
}

// 执行
runAllModels();


就这么简单。你不需要去豆包官方拉SDK，不用去研究GPT和DeepSeek的接口文档，把 `model` 字段一换，程序就能自动切换模型进行对话。这就是“一键接入”的真正含义。
