    // 关键改动 3：国内直连，不需要 proxy 配置
});

async function main() {
    const response = await client.chat.completions.create({
        model: 'o3-mini',   // 模型名称直接填，千聚完全兼容
        messages: [
            { role: 'user', content: '请用中文解释一下区块链的共识机制，并给出一个 Node.js 的简单代码示例。' }
        ],
        // o3-mini 建议开启 reasoning_effort 来平衡质量与成本
        reasoning_effort: 'medium', // 可选值: 'low', 'medium', 'high'
        max_completion_tokens: 4096,

    console.log('回答:', response.choices[0].message.content);
    console.log('Token 用量:', {
        input: response.usage?.prompt_tokens,
        output: response.usage?.completion_tokens,
        total: response.usage?.total_tokens
    });
}

main();
