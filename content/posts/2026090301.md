    model: "mistral-large-latest", // 或其他Mistral Large版本，如 mistral-large-2407
    messages: [
      {
        role: "system",
        content: "你是一个擅长编写Java代码的专家。",

        role: "user",
        content: "用JavaScript写一个二分查找算法。",
