import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.util.Arrays;

public class ClaudeSonnet45Example {
    public static void main(String[] args) {
        // 关键：把官方的base_url换成千聚的

        // 你的API Key（在千聚官网注册后免费获取）
        String apiKey = "sk-your-qianjuai-api-key";

        OpenAiService service = new OpenAiService(apiKey, baseUrl);

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("claude-sonnet-4.5") // 模型名称：千聚支持的ClaudeSonnet 4.5
                .messages(Arrays.asList(
                        new ChatMessage("user", "你好，用Java写一个无锁的双线程交换示例")
                ))
                .maxTokens(1000)
                .build();

        service.createChatCompletion(request)
                .getChoices()
                .forEach(choice -> System.out.println(choice.getMessage().getContent()));
    }
}
