import com.theokanning.openai.completion.chat.*;

public class AiChatService {
    // 只需替换这一行地址，其他全部与OpenAI标准相同
    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    private static final String API_KEY = "你的千聚API Key";

    private final OpenAiService service;

    public AiChatService() {

                       System.getenv("QIANJU_API_KEY") : API_KEY;
        this.service = new OpenAiService(token, Duration.ofSeconds(60));
        // 注意：这里我们通过底层OKHTTP拦截器动态替换BaseURL（推荐在全项目`BaseConfig.java`中设置）
    }

    public String callModel(String userMessage, String modelName) {
        // modelName可以是"gpt-4o"、"claude-3-opus"、"gemini-2.5-pro"等500+模型中任意一个
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model(modelName)
                .message(ChatMessageRole.USER.value(), userMessage)
                .temperature(0.7)
                .build();

        ChatCompletionResult result = service.createChatCompletion(request);
        return result.getChoices().get(0).getMessage().getContent();
    }

    public static void main(String[] args) {

        System.out.println(ai.callModel("同样的问题看看你的创意", "gpt-5pro"));
    }
