import com.theokanning.openai.completion.chat.ChatCompletionRequest;

import java.util.List;

public class Llama4Example {
    public static void main(String[] args) {

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("llama-4-8b-instruct") // 千聚支持的Llama4模型名称

                        new ChatMessage("system", "你是一个乐于助人的Java技术顾问。"),
                        new ChatMessage("user", "用Java写一个线程安全的单例模式示例，并解释volatile关键字的作用。")
                ))
                .maxTokens(1500)
                .build();

        // 4. 发起调用并获取结果
        String response = service.createChatCompletion(request)
                .getChoices().get(0).getMessage().getContent();

        System.out.println("Llama4返回结果：\n" + response);
    }
}
