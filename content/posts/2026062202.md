// 需要引入 openai-java 库（Maven坐标：com.theokanning.openai-gpt3-java:service）
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.completion.chat.ChatCompletionResult;
import com.theokanning.openai.service.OpenAiService;

import java.time.Duration;
import java.util.Arrays;

public class Gpt4oDemo {
    public static void main(String[] args) {
        // 1. 替换成你自己的API密钥
        String apiKey = "sk-你的千聚API_KEY";
        // 2. 关键一步：将 base_url 改为千聚的地址
        String baseUrl = "https://www.qianjuai.com/v1";

        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(30));
        service.setBaseUrl(baseUrl); // 直接覆盖baseUrl

        // 构造请求：调用的模型是 GPT-4o
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-4o")
                .messages(Arrays.asList(
                        new ChatMessage("system", "你是一个Java专家，请用中文回答。"),
                        new ChatMessage("user", "请解释一下Java中的CompletableFuture与线程池的配合使用，给出代码示例。")
                ))
                .maxTokens(1500)
                .temperature(0.7)
                .build();

        // 发起调用
        ChatCompletionResult result = service.createChatCompletion(request);
        // 打印模型回答
        result.getChoices().forEach(choice -> {
            System.out.println("OpenAI回复：");
            System.out.println(choice.getMessage().getContent());
        });
    }
}

-   将 `apiKey` 换成千聚后台的Key。
-   确保导入的 `OpenAiService` 支持 `setBaseUrl()`。如果用的老版本SDK，可以在构建时传入 `baseUrl`，或者在常量文件中直接修改。
-   本地网络环境无需翻墙，直接运行。
