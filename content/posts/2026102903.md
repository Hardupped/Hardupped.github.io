## 豆包接入Java示例：真的只需要改一行代码

### 2. 完整Java Demo

真的，只需要把`baseUrl`从OpenAI的官方地址换成千聚的接口，把API Key换成你在千聚申请的Key，就可以了。

java
import okhttp3.*;
import org.json.JSONArray;
import org.json.JSONObject;

import java.io.IOException;

public class DoubaoDemo {

    // 隐藏通道的关键：替换baseUrl
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    private static final String API_KEY = "sk-你的千聚API Key"; // 在控制台获取

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient().newBuilder()
                .build();

        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "doubao-pro-32k"); // 指定豆包模型
        requestBody.put("max_tokens", 1024);
        requestBody.put("stream", false);

        JSONArray messages = new JSONArray();
        JSONObject userMessage = new JSONObject();
        userMessage.put("role", "user");
        userMessage.put("content", "用Java写一个单例模式的例子");
        messages.put(userMessage);
        requestBody.put("messages", messages);

        MediaType mediaType = MediaType.parse("application/json");
        Request request = new Request.Builder()
                .url(API_URL)
                .method("POST", RequestBody.create(mediaType, requestBody.toString()))
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .build();

        Response response = client.newCall(request).execute();
        String responseBody = response.body().string();
        System.out.println("豆包回复： " + responseBody);

        // 解析JSON，提取content
        JSONObject jsonResponse = new JSONObject(responseBody);
        String content = jsonResponse.getJSONArray("choices")
                .getJSONObject(0)
                .getJSONObject("message")
                .getString("content");
        System.out.println("\n--- 提取的文本内容 ---");
        System.out.println(content);
    }
}
