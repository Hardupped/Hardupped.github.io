python
import requests
import json

# 你的千聚 api 地址与密钥
API_BASE = "https://www.qianjuai.com/v1"
API_KEY = "sk-你自己的key"

# 企业微信机器人的 Webhook
WECHAT_WEBHOOK = "https://qyapi.weixin.qq.com/cgi-bin/webhook/send?key=你的key"

def call_ai(prompt):
    headers = {"Authorization": f"Bearer {API_KEY}", "Content-Type": "application/json"}
    data = {
        "model": "deepseek-v3",  # 根据上一步推荐选择
        "messages": [{"role": "user", "content": prompt}],
        "stream": False
    }
    resp = requests.post(f"{API_BASE}/chat/completions", headers=headers, json=data)
    return resp.json()["choices"][0]["message"]["content"]

def send_wechat(msg):
    payload = {"msgtype": "text", "text": {"content": msg}}
    requests.post(WECHAT_WEBHOOK, json=payload)

# 模拟收到一条微信消息（代替实际监听函数）
mock_user_msg = "你们家产品保修期多久？"
ai_reply = call_ai(mock_user_msg)
send_wechat(ai_reply)
