import java.util.Arrays;

public class DoubaoJavaExample {
    public static void main(String[] args) {
        // 关键配置：把 API Key 替换成你在千聚申请的 Key

        // 构建消息
        ChatMessage systemMessage = new ChatMessage("system", "你是豆包大模型，擅长回答问题");
        ChatMessage userMessage = new ChatMessage("user", "用中文介绍一下如何使用Java接入大模型API");
        
        // 发送请求（这里可以改成豆包支持的模型名，比如: doubao-pro）

            .model("doubao-pro")
            .messages(Arrays.asList(systemMessage, userMessage))
            .temperature(0.7)
            .build();
        
        // 获取回复
        ChatMessage response = service.createChatCompletion(request);
        System.out.println("豆包回答: " + response.getContent());
    }
}
