import openai

def batch\_process\_with\_deepseek(questions: list) -> list:
    """
    批量处理多个问题，合并成一次 API 调用，节省 Token 开销。
    """
    # 构造一个长 prompt，包含所有问题
    combined\_prompt = "请一次性回答下面所有 Python 代码问题。每个答案保持在 2 句话以内。不要思考，不要输出解析。\n\n"
    for idx, question in enumerate(questions):
        combined\_prompt += f"问题 {idx+1}: {question}\n"

    # 一次调用，产出多个答案
    response = client.chat.completions.create(
        model="deepseek-r1",
        messages=[
            {"role": "system", "content": "你是顶级助手，直接输出答案列表，不输出推理过程。"},
            {"role": "user", "content": combined\_prompt}
        ],
        max\_tokens=2048,
        temperature=0.1
    )
    answer_text = response.choices[0].message.content

    # 简单分割，假设模型按编号输出
    answers = answer\_text.split("\\n")
    return [a for a in answers if a.strip()]

# 实际使用
questions = [
    "如何用 Python 读取 JSON 文件？",
    "请写一个装饰器打印函数执行时间",
    "解释 Python 中 'is' 与 '==' 的区别"
]
result = batch\_process\_with\_deepseek(questions)
print("生成的回答组合：", result)
