        // 抖音上那些但凡让你交钱买库的卖家，都不会把这个地址告诉你

        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(30, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        JSONObject jsonRequest = new JSONObject();
        jsonRequest.put("model", "grok-3"); // 关键：调用Grok-3模型
        jsonRequest.put("messages", new JSONArray()
                .put(new JSONObject().put("role", "system").put("content", "你是一个Java专家"))
                .put(new JSONObject().put("role", "user").put("content", "用Grok3写出完整的Java调用示例")));

        Request request = new Request.Builder()
                .url(apiUrl)
                .addHeader("Authorization", "Bearer " + apiKey)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(jsonRequest.toString(), MediaType.get("application/json; charset=utf-8")))
                .build();

        try {
            Response response = client.newCall(request).execute();
            System.out.println(response.body().string());
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
